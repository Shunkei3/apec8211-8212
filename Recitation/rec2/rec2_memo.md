$$\begin{align*}
E[X] = \begin{cases}
\sum_{j=1}^{\infty} \tau_j \pi_j & \text{for discrete } X\\
\int_{-\infty}^\infty x f(x) \,dx & \text{for continuous } X
\end{cases}
\end{align*}$$


$$\begin{align}
Var[X] = \begin{cases}
\sum_{j=1}^{\infty} (\tau_j - E[X])^2 \pi_j & \text{for discrete } X\\
\int_{-\infty}^\infty (x - E[X])^2 f(x) \,dx & \text{for continuous } X
\end{cases}
\end{align}$$
